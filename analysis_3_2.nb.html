Dissertation 3.2 Analysis (Oddball Microbiologist)

Super similar to the 3.1 analysis, but with oddball exceptions. Rule-followers were categorized according to one dimension; oddballs (introduced in LB2) were categorized according to an XOR rule. Participants uncovered parts of the stimulus with their cursor, and their cursor position was recorded as a proxy for eye-tracking data.

# Okay now get proportion per trial; start with fixation duration
df_coords <- df_coords %>%
  filter(!(sona_id %in% exclusions)) %>%
  filter(block != "Other") %>%
  arrange(sona_id, time) %>%
  group_by(sona_id, trial_index) %>%
  mutate(trial_dur = last(time) - first(time)) %>%
  ungroup() %>%
  arrange(sona_id, trial_index, time) %>%
  group_by(sona_id, trial_index) %>%
  mutate(duration = lead(time, default = last(time)) - time) %>%
  replace_na(list(duration = 0)) %>%
  ungroup()

df_coords <- df_coords %>% 
  group_by(sona_id, block) %>%
  mutate(runningCount = dense_rank(trial_index))

# Now sum for each dimension and get proportion
df_coords <- df_coords %>% ungroup() %>%
  group_by(sona_id, trial_index, dim) %>%
  mutate(dim_dur = sum(duration)) %>%
  mutate(dim_prop = dim_dur/trial_dur) %>%
  ungroup() %>%
  group_by(sona_id, trial_index, block, dim) %>%
  summarize(mean_prop = mean(dim_prop))

# And I want a wider format with no NAs before I merge:
df_coords <- df_coords %>% 
  pivot_wider(names_from = dim, values_from = mean_prop)

df_coords <- df_coords %>%
  mutate(across(c(D1, D2, D3), ~ifelse(is.na(.), 0, .)))
